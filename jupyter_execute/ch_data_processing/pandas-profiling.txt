# Pandas profiling

import pandas as pd
from pandas_profiling import ProfileReport

df_train = pd.read_csv("../data/titanic/train.csv")

profile = ProfileReport(df_train, title="Pandas Profiling Report for Titanic train dataset")
profile.to_file("../data/titanic/train_profiling.html")
profile.to_widgets()
